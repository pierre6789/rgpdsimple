Code HTML du bandeau cookies

À intégrer sur votre site (par exemple dans le fichier de layout global) :

<div id="cookie-banner" style="position:fixed;bottom:0;left:0;right:0;background:#111827;color:#ffffff;padding:12px 16px;font-size:14px;z-index:9999;display:flex;flex-wrap:wrap;align-items:center;justify-content:space-between;">
  <span style="margin-bottom:8px;">
    {{nomEntreprise}} utilise des cookies pour le fonctionnement du site et la mesure d'audience. 
    Vous pouvez accepter ou refuser les cookies non essentiels.
  </span>
  <div>
    <button onclick="acceptCookies()" style="margin-right:8px;padding:8px 12px;border-radius:999px;border:none;background:#10b981;color:#ffffff;cursor:pointer;">Tout accepter</button>
    <button onclick="refuseCookies()" style="padding:8px 12px;border-radius:999px;border:1px solid #e5e7eb;background:#111827;color:#ffffff;cursor:pointer;">Refuser</button>
  </div>
</div>
<script>
  function setCookie(name, value, days) {
    var expires = "";
    if (days) {
      var date = new Date();
      date.setTime(date.getTime() + (days*24*60*60*1000));
      expires = "; expires=" + date.toUTCString();
    }
    document.cookie = name + "=" + (value || "")  + expires + "; path=/";
  }
  function hideBanner() {
    var banner = document.getElementById("cookie-banner");
    if (banner) banner.style.display = "none";
  }
  function acceptCookies() {
    setCookie("cookie_consent", "accepted", 180);
    hideBanner();
    // Initialiser ici vos outils de mesure d'audience / marketing
  }
  function refuseCookies() {
    setCookie("cookie_consent", "refused", 180);
    hideBanner();
  }
  (function() {
    if (document.cookie.indexOf("cookie_consent=") !== -1) {
      hideBanner();
    }
  })();
</script>
